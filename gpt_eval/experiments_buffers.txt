    ap.add_argument("--alpha_hi", type=float, default=8.0, help="Initial upper bracket for auto-alpha.")
    ap.add_argument("--alpha_eff", type=float, default=0.7, help="Manual alpha if --auto_alpha is off.")
    ap.add_argument("--steer_gain", type=float, default=None, help="Override steerer.steer_gain.")
    ap.add_argument("--alpha_mode", type=str, default=None, choices=["abs", "frac"])

    # injection / layout
    ap.add_argument("--injection_point", type=str, default=None, choices=["post", "mha", "mlp", "final_norm"])
    ap.add_argument("--last_only", action="store_true", help="Inject only at last position (more stable, weaker visibly).")

    # layer mix
    ap.add_argument("--k_runtime", type=int, default=1)
    ap.add_argument("--max_layers", type=int, default=2)
    ap.add_argument("--min_weight", type=float, default=0.10)
    ap.add_argument("--prior_boost", type=float, default=0.10)
    ap.add_argument("--temperature_sel", type=float, default=0.50, help="Softmax temperature for layer weighting.")
    ap.add_argument("--layer_policy", type=str, default="prefer_verified",
                    choices=["prefer_verified", "force_verified", "runtime_only"])
    ap.add_argument("--min_verified_weight", type=float, default=0.50,
                    help="Floor for total weight allocated to verified layers when using prefer_verified.")
    ap.add_argument("--signed_filter_alpha", type=float, default=2.0,
                    help="Tiny alpha used to test per-layer signed KL bias.")

    # decoding profile
    ap.add_argument("--style", type=str, default="balanced",
                    choices=["balanced", "strong"],
                    help="balanced = higher quality; strong = more visibly steered.")

    # logging
    ap.add_argument("--log_level", type=str, default="info", choices=["silent", "warn", "info", "debug"])

    args = ap.parse_args()

    # Load steerer (reads artifacts.pkl; sets verified layers if present)
    steerer = load_steerer(args.results_dir)
    if args.steer_gain is not None:
        steerer.steer_gain = float(args.steer_gain)
    if args.injection_point is not None:
        steerer.injection_point = args.injection_point
    if args.alpha_mode is not None:
        steerer.alpha_mode = args.alpha_mode
    # IMPORTANT: default to False for stronger visible effect; user can opt-in to last_only.
    steerer.last_position_only = bool(args.last_only)
    steerer.log_level = args.log_level

    # fast sanity — should visibly move logits when cranked
    assert_hook_moves_logits(steerer, "Say hello in one word.", args.trait, intensity=6.0)

    # ---- (1) initial hybrid selection
    layers, weights, norms = select_layers_for_prompt(
        steerer, args.prompt, args.trait, args.intensity,
        system=args.system, k_runtime=args.k_runtime,
        prior_boost=args.prior_boost, temperature=args.temperature_sel,
        max_layers=args.max_layers, min_weight=args.min_weight
    )

    verified = getattr(steerer, "_trait_layers", {}).get(args.trait.lower(), [])
    top_sorted = sorted(norms.items(), key=lambda kv: kv[1], reverse=True)

    # Policy overrides
    if args.layer_policy == "force_verified" and verified:
        layers = list(verified)[:args.max_layers]
        weights = [1.0 / len(layers)] * len(layers)
    elif args.layer_policy == "runtime_only":
        layers = [L for (L, _) in top_sorted[:args.max_layers]]
        weights = [1.0 / len(layers)] * len(layers)
    # else keep hybrid result

    # ---- (2) decide global sign using the *verified* anchor when available
    if verified:
        with SteerConfigPatch(steerer, list(verified)[:args.max_layers],
                              [1.0/len(list(verified)[:args.max_layers])] * len(list(verified)[:args.max_layers])):
            sgn = steerer._calibrate_polarity(args.trait)
    else:
        with SteerConfigPatch(steerer, layers, weights):
            sgn = steerer._calibrate_polarity(args.trait)

    # ---- (3) signed-bias filtering of runtime candidates (prevents opposite polarity)
    if args.layer_policy != "force_verified" and args.k_runtime > 0:
        # candidates from top Δlogits that are NOT verified
        runtime_pool = [L for (L, _) in top_sorted if L not in set(verified)]
        filtered = []
        for L in runtime_pool:
            klp, kln = _kl_signed_on_layer(
                steerer, args.prompt, args.trait, L, system=args.system,
                alpha_test=args.signed_filter_alpha
            )
            # keep only layers that favor the chosen global sign
            if (sgn > 0 and klp >= kln) or (sgn < 0 and kln >= klp):
                filtered.append(L)
        # rebuild mix: verified ∪ filtered[:max_layers - len(verified)]
        new_layers = list(verified)
        for L in filtered:
            if len(new_layers) >= args.max_layers:
                break
            if L not in new_layers:
                new_layers.append(L)
        if new_layers:
            layers = new_layers
            weights = [1.0 / len(layers)] * len(layers)

    # ---- (4) enforce a floor for verified weights (keeps anchor strong)
    if args.layer_policy == "prefer_verified" and verified:
        weights = _enforce_verified_floor(layers, weights, verified, args.min_verified_weight)

    print("\n[Hybrid selection]")
    print(f"  verified layers (from artifacts): {verified if verified else '[]'}")
    print("  top Δlogits norms:", {L: round(v, 3) for L, v in top_sorted[:8]})
    print(f"  chosen layers: {layers}")
    print(f"  weights: { [round(w, 3) for w in weights] }")

    print(f"\n[polarity][signed-probe] trait={args.trait} (anchor={'verified' if verified else 'mix'}) -> sign={'+1' if sgn>0 else '-1'}")

    # ---- (5) run inside the final layer mix
    with SteerConfigPatch(steerer, layers, weights):
        # stronger visibility if not last_only
        steerer.last_position_only = bool(args.last_only)
        # zero-centering can help through RMSNorm; harmless if unsupported
        steerer.zero_center_delta = True

        # auto alpha (compat for older calibrators without alpha_hi)
        if args.auto_alpha:
            try:
                alpha_pos = calibrate_alpha_for_sign_in_context(
                    steerer, args.prompt, args.trait, +1,
                    target_kl=args.target_kl, system=args.system,
                    alpha_hi=args.alpha_hi
                )
                alpha_neg = calibrate_alpha_for_sign_in_context(
                    steerer, args.prompt, args.trait, -1,
                    target_kl=args.target_kl, system=args.system,
                    alpha_hi=args.alpha_hi
                )
            except TypeError:
                alpha_pos = calibrate_alpha_for_sign_in_context(
                    steerer, args.prompt, args.trait, +1,
                    target_kl=args.target_kl, system=args.system
                )
                alpha_neg = calibrate_alpha_for_sign_in_context(
                    steerer, args.prompt, args.trait, -1,
                    target_kl=args.target_kl, system=args.system
                )
            # sanity print under same mix
            kpos = _next_token_kl(steerer, args.prompt, args.trait, alpha_pos, +1, system=args.system)
            kneg = _next_token_kl(steerer, args.prompt, args.trait, alpha_neg, -1, system=args.system)
            print(f"[alpha] auto (+/-): {alpha_pos:.4f} / {alpha_neg:.4f} | KL(+/−): {kpos:.3f} / {kneg:.3f}")
        else:
            alpha_pos = alpha_neg = float(args.alpha_eff)

        # decode profile
        if args.style == "balanced":
            temp, top_p, top_k, rep = 0.4, 0.85, 50, 1.07
        else:  # strong
            temp, top_p, top_k, rep = 0.7, 0.9, 50, 1.02

        # paired generations
        base_txt, up_txt = paired_generations(
            steerer, args.prompt, args.trait,
            alpha_signed=(+alpha_pos / float(steerer.steer_gain)),
            max_new_tokens=180, system=args.system,
            temperature=temp, top_p=top_p, top_k=top_k, repetition_penalty=rep
        )
        _, down_txt = paired_generations(
            steerer, args.prompt, args.trait,
            alpha_signed=(-alpha_neg / float(steerer.steer_gain)),
            max_new_tokens=180, system=args.system,
            temperature=temp, top_p=top_p, top_k=top_k, repetition_penalty=rep
        )

    print("\n=== Baseline ===\n", base_txt.strip())
    print(f"\n=== Steered (+{args.trait} @ α={alpha_pos:.3f}) ===\n", up_txt.strip())
    print(f"\n=== Steered (-{args.trait} @ α={alpha_neg:.3f}) ===\n", down_txt.strip())

@torch.no_grad()
def _kl_signed_on_layer(steerer, prompt: str, trait: str, layer: int, *, system: str = None,
                        alpha_test: float = 2.0) -> Tuple[float, float]:
    """
    Measure next-token KL when steering only this layer with a small +/− alpha.
    Returns (KL_plus, KL_minus).
    """
    with SteerConfigPatch(steerer, [layer], [1.0]):
        # use the same KL routine as auto-alpha does, but signed intensity
        klp = _next_token_kl(steerer, prompt, trait, alpha_test, +1, system=system)
        kln = _next_token_kl(steerer, prompt, trait, alpha_test, -1, system=system)
    return float(klp), float(kln)


def _enforce_verified_floor(layers: List[int], weights: List[float],
                            verified: List[int], min_verified_weight: float) -> List[float]:
    """
    Ensure the sum of weights on verified layers is at least 'min_verified_weight'.
    If needed, rescale runtime weights proportionally.
    """
    if not verified or min_verified_weight <= 0.0:
        return weights
    vset = set(verified)
    w = np.array(weights, dtype=np.float32)
    v_sum = float(np.sum([w[i] for i, L in enumerate(layers) if L in vset]))
    if v_sum >= min_verified_weight:
        return weights
    # increase verified portion to floor; reduce others proportionally
    scale_up = (min_verified_weight / max(v_sum, 1e-9))
    w_new = w.copy()
    for i, L in enumerate(layers):
        if L in vset:
            w_new[i] = w[i] * scale_up
    # renormalize the remainder
    rem_target = 1.0 - min_verified_weight
    rem_idxs = [i for i, L in enumerate(layers) if L not in vset]
    rem_sum = float(np.sum(w[rem_idxs])) if rem_idxs else 0.0
    if rem_idxs and rem_sum > 0:
        for i in rem_idxs:
            w_new[i] = (w[i] / rem_sum) * rem_target
    # final normalize
    s = float(np.sum(w_new))
    if s > 0:
        w_new = (w_new / s).tolist()
    else:
        w_new = weights
    return w_new
